char readSign()
{

        while (true)
        {
                getline(cin, data);

                if (data.length() == 1 )
                {
                        sign = data[0];

void loadUsersFromFile (vector <User> &userList)
{
        string data = "";
        int position = 1;

        fstream plik;
        plik.open(userFile.c_str(), ios::in);

        if (plik.good() == true)

                {
                        switch (position)
                        {
                        case 1:
                                users.id = atoi(data.c_str());
                                break;
                        case 2:
                                users.username = data;
                                break;
                        case 3:
                                users.password = data;
                                break;
                        }
                        if (position >= 3)
                        {
                                userList.push_back(users);
                                position = 1;
                        }
                        else
                        {
                                position++;
                        }
                }
                plik.close();
        }
	else
	{
		cout << "Nie uzyskano dostepu do pliku." << endl;
	}
        return;
}
